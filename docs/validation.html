import re
from string import ascii_lowercase


class Valid:
    "Set of functions that validate the different parts of the Homie Convention"

    @classmethod
    def topic(cls, topic: str) -> bool:
        """`SPEC`

        An MQTT topic consists of one or more topic levels,
        separated by the slash character (/).
        """
        raise NotImplementedError

    @staticmethod
    def topic_id(topic_id: str) -> bool:
        """`SPEC`

        A topic level ID MAY contain lowercase letters from a to z,
        numbers from 0 to 9 as well as the hyphen character (-).

        A topic level ID MUST NOT start or end with a hyphen (-).
        The special character $ is used and reserved for Homie attributes.
        """
        return topic_id[0] != "-" and all(
            c == "-" or c in ascii_lowercase for c in topic_id
        )

    @staticmethod
    def payload(payload: str) -> bool:
        """`SPEC`

        Every MQTT message payload MUST be sent as a UTF-8 encoded string.

        The value published as payload MUST be valid for the
        respective property/attribute type as per the list below.
        """
        # TODO: Change payload to bytes  # pylint: disable=fixme
        return isinstance(payload, str)

    @classmethod
    def string(cls, string_payload: str) -> bool:
        """`SPEC`

        String types are limited to 268,435,456 characters
        An empty string ("") is a valid payload
        """
        return cls.payload(string_payload) and len(string_payload) < 2 ** 28

    @classmethod
    def integer(cls, integer_payload: str) -> bool:
        """`SPEC`

        Integer types are UTF-8 encoded string literal
        representations of 64-bit signed whole numbers.

        Integers range from -9,223,372,036,854,775,808 (-2^63)
        to 9,223,372,036,854,775,807 (263-1).

        The payload may only contain whole numbers and the negation
        character `-`. No other characters including spaces (` `) are permitted.

        A string with just a negation sign (`-`) is not a valid payload.

        An empty string (``) is not a valid payload.
        """
        if not bool(integer_payload):
            return False

        if not cls.payload(integer_payload):
            return False

        integer_match = r"^-?(\d+)"
        if re.match(integer_match, integer_payload) is None:
            return False

        number = int(integer_payload)
        if number < -(2 ** 63) or number > 2 ** 63:
            return False

        return True

    @classmethod
    def float(cls, float_payload: str) -> bool:
        """`SPEC`

        Float types are UTF-8 encoded string literal representations
        of 64-bit signed floating point numbers.

        Floats range from  2^(-1074) to (2-(2^-52))*2^1023.
        The payload may only contain whole numbers, the negation character “-",
        the exponent character “e” or “E” and the decimal separator “.",
        no other characters, including spaces (” “) are permitted

        The dot character (".") is the decimal separator (used if necessary)
        and may only have a single instance present in the payload.

        Representations of numeric concepts such as “NaN” (Not a Number)
        and “Infinity” are not a valid payload.

        A string with just a negation sign ("-") is not a valid payload
        An empty string ("") is not a valid payload.
        """
        if not bool(float_payload):
            return False

        if not cls.payload(float_payload):
            return False

        float_match = r"^-?(\d+)(.\d+)?([eE]\d+)?$"
        if re.match(float_match, float_payload) is None:
            return False

        return True

    @classmethod
    def boolean(cls, boolean_payload: str) -> bool:
        """`SPEC`

        Booleans must be converted
        to the string literals "true" or "false".

        Representation is case sensitive, e.g. "TRUE" or "FALSE" are not valid payloads.

        An empty string ("") is not a valid payload.
        """
        if not bool(boolean_payload):
            return False

        if not cls.payload(boolean_payload):
            return False

        if boolean_payload != "true" or boolean_payload != "false":
            return False

        return True

    @classmethod
    def enum(cls, enum_payload: str, format_of: str) -> bool:
        """`SPEC`

        Enum payloads must be one of the values specified
        in the format definition of the property.

        Enum payloads are case sensitive, e.g. "Car" will
        not match a format definition of "car".

        Payloads should have leading and trailing whitespace removed.
        An empty string ("") is not a valid payload.
        """
        if not bool(enum_payload) or not bool(format_of):
            return False

        if not cls.payload(enum_payload):
            return False

        if enum_payload not in format_of.split(","):
            return False

        if enum_payload != enum_payload.strip():
            return False

        return True

    @classmethod
    def color(cls, color_payload: str, format_of: str) -> bool:
        """`SPEC`

        Color payload validity varies depending on the
        property format definition of either "rgb" or "hsv".

        Both payload types contain comma separated
        whole numbers of differing restricted ranges.

        The encoded string may only contain whole numbers and
        the comma character ",", no other characters are permitted,
        including spaces (" ").

        Payloads for type "rgb" contains 3 comma separated values
        of numbers with a valid range between 0 and 255. e.g. 100,100,100 .

        Payloads for type "hsv" contains 3 comma separated values of numbers.
        The first number has a range of 0 to 360,
        the second and third numbers have a range of 0 to 100. e.g. 300,50,75.

        An empty string ("") is not a valid payload
        """
        if (
            not bool(color_payload)
            or not bool(format_of)
            or not cls.payload(color_payload)
        ):
            return False

        color_match = r"(\d+),(\d+),(\d+)"
        match = re.match(color_match, color_payload)
        if match is None:
            return False

        values = [int(c) for c in match.groups()]
        if format_of == "rgb":
            if any(v < 0 or v > 255 for v in values):
                return False

        elif format_of == "hsv":
            if values[0] < 0 or values[0] > 360 or any(v < 0 or v > 100 for v in values):
                return False

        else:
            return False

        return True
